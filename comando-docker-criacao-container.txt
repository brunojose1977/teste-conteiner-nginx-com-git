#Comandos da criacao do conteiner

#Criação da imagem
sudo docker build --tag my-nginx-image .

##Configurações do conteiner my-nginx
clear &&
sudo docker run --name my-nginx --volume my-volume:/usr/share/nginx/html:ro -it -d -p 8081:80 my-nginx-image &&
sudo docker start my-nginx &&
sudo docker exec -it my-nginx ./my-nginx-container-bootstrap.sh &&
sudo docker exec -it my-nginx bash

# Comando para parar e destruir o conteiner
sudo docker rm -f my-nginx

#Comando para destruir a imagem
sudo docker rmi my-nginx-image
